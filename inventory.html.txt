<script>
const form = document.getElementById("restockForm");
const status = document.getElementById("status");
const inventoryList = document.getElementById("inventoryList");

form.addEventListener("submit", async (e) => {
    e.preventDefault();

    status.textContent = "";
    status.className = "";

    const data = {
        location: document.getElementById("location").value,
        machine: document.getElementById("machine").value,
        product: document.getElementById("product").value,
        amountAdded: parseInt(document.getElementById("amountAdded").value),
        whoRestocked: document.getElementById("whoRestocked").value,
        notes: document.getElementById("notes").value
    };

    // Get GPS if available
    if (navigator.geolocation) {
        navigator.geolocation.getCurrentPosition(
            (position) => {
                data.latitude = position.coords.latitude;
                data.longitude = position.coords.longitude;
                submitRestock(data);
            },
            () => {
                // GPS denied or unavailable
                submitRestock(data);
            }
        );
    } else {
        submitRestock(data);
    }
});

async function submitRestock(data) {
    try {
        const response = await fetch("https://cc4x8ok8s8.execute-api.us-east-1.amazonaws.com/dev/add-restock", {
            method: "POST",
            headers: { "Content-Type": "application/json" },
            body: JSON.stringify(data)
        });

        if (!response.ok) {
            const err = await response.json();
            throw new Error(err.message || "Failed to log restock");
        }

        status.textContent = "Restock logged successfully!";
        status.className = "success";

        form.reset();
        loadInventory();

    } catch (err) {
        status.textContent = "Error: " + err.message;
        status.className = "error";
    }
}

async function loadInventory() {
    try {
        const response = await fetch("https://cc4x8ok8s8.execute-api.us-east-1.amazonaws.com/dev/inventory", {
            method: "GET"
        });

        if (!response.ok) throw new Error("Failed to load inventory");

        const entries = await response.json();

        inventoryList.innerHTML = entries.map(entry => `
            <div class="entry">
                <strong>${entry.location} - ${entry.machine} - ${entry.product}</strong><br>
                Added: ${entry.amountAdded} on ${new Date(entry.restockedAt).toLocaleString()}<br>
                By: ${entry.whoRestocked}${entry.notes ? `<br>Notes: ${entry.notes}` : ''}<br>
                ${entry.latitude ? `Location: ${entry.latitude.toFixed(4)}, ${entry.longitude.toFixed(4)}` : ''}
            </div>
        `).join('') || "<p>No restocks logged yet.</p>";

    } catch (err) {
        inventoryList.innerHTML = "<p>Error loading inventory: " + err.message + "</p>";
    }
}

// Load recent restocks on page load
loadInventory();
</script>
